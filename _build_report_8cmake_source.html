49 FUNCTION(INTERNAL_CREATE_BUILD_REPORT)
50  IF(CONFIG_CREATE_BUILD_REPORT)
51  IF(NOT "${DOXYGEN_EXECUTABLE}" STREQUAL "DOXYGEN_EXECUTABLE-NOTFOUND")
52  MESSAGE("")
53  MESSAGE(---------------------------------------------------------------------------)
54  MESSAGE("Creating build report...")
55  IF(EXISTS ${DOXYGEN_EXECUTABLE})
56  SET(path_to_build_report "${ACME_PATH}/plugins/BuildReport")
57  SET(path_to_doc "${CMAKE_DOC_OUTPUT_DIRECTORY}")
58  SET(doxygen_run_directory "${PROJECT_SOURCE_DIR}/modules")
59 
60  CONFIGURE_FILE(${path_to_build_report}/doxyfile_build_report_template ${path_to_doc}/doxyfile_build_report_${PROJECT_NAME})
61  CONFIGURE_FILE(${path_to_build_report}/0_mainpagetemplate.dox ${path_to_doc}/0_mainpage.dox)
62  CONFIGURE_FILE(${path_to_build_report}/1_full_dependency_graph_template.dox ${path_to_doc}/1_full_dependency_graph.dox)
63 
64  EXEC_PROGRAM(${DOXYGEN_EXECUTABLE} ${doxygen_run_directory}
65  ARGS ${path_to_doc}/doxyfile_build_report_${PROJECT_NAME}
66  )
67  IF(EXISTS ${path_to_doc}/dependency_graph.map)
68  FILE(READ ${path_to_doc}/dependency_graph.map map_content)
69  FILE(APPEND ${path_to_doc}/html/_full_01dependency_01graph.html \n${map_content})
70 
71  SET(highlight_content \n<script\ type="text/javascript"\ src="jquery.maphilight.js"></script>\n<script\ type="text/javascript">$(function()\ {\n$('.map').maphilight();\n});\n</script>)
72  FILE(APPEND ${path_to_doc}/html/_full_01dependency_01graph.html ${highlight_content})
73 
74  CONFIGURE_FILE(${path_to_doc}/dependency_graph.png ${path_to_doc}/html/dependency_graph.png COPYONLY)
75  CONFIGURE_FILE(${path_to_build_report}/jquery.maphilight.js ${path_to_doc}/html/jquery.maphilight.js COPYONLY)
76  ENDIF()
77 
78  MESSAGE(STATUS "Build report successfully created!")
79  ELSE()
80  MESSAGE(STATUS "WARNING: The doxygen executable ${DOXYGEN_EXECUTABLE} doesn't exist.")
81  MESSAGE(STATUS "WARNING: The build report was not created.")
82  ENDIF()
83  ENDIF()
84  ENDIF()
85 ENDFUNCTION(INTERNAL_CREATE_BUILD_REPORT)

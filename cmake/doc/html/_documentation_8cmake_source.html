17 # Adds a Documentation target to the project, if a "doc" folder exists.
18 # This doc folder must contain a doxygen configuration named Doxyfile.in
19 # The documentation will be generated into the deliverable folder.
20 
21 
22 #--------------------------------------------------------------------------
23 # Plugin API
24 #--------------------------------------------------------------------------
25 
26 SET(CONFIG_OPEN_DOCUMENTATION_FILE 0 CACHE BOOL "immediately opening index.html when building documentation")
27 MARK_AS_ADVANCED(CONFIG_OPEN_DOCUMENTATION_FILE)
28 SET(CONFIG_DOCUMENTATION_FOLDER_NAME "documentation" CACHE STRING "folder name for the documentation folder")
29 MARK_AS_ADVANCED(CONFIG_DOCUMENTATION_FOLDER_NAME)
30 
31 
32 #--------------------------------------------------------------------------
33 # Plugin External Tools
34 #--------------------------------------------------------------------------
35 
36 PRINT_DETAILS(STATUS "...requires external tool Doxygen")
37 IF(EXISTS ${CMAKE_CURRENT_SOURCE_DIR}/doc/)
38  IF(NOT DOXYGEN_FOUND)
39  INCLUDE(${ACME_PATH}/tools/FindDoxygen.cmake)
40  ENDIF()
41 ENDIF()
42 
43 
44 #--------------------------------------------------------------------------
45 # Plugin Functions
46 #--------------------------------------------------------------------------
47 
48 IF(EXISTS ${CMAKE_CURRENT_SOURCE_DIR}/doc/)
49  #find_package(Doxygen)
50  IF(DOXYGEN_FOUND)
51  # show the sources in the target
52  FILE(GLOB_RECURSE doc_files "${CMAKE_CURRENT_SOURCE_DIR}/doc/*.*") # TODO how to keep the file directory structure?
53  SOURCE_GROUP("Documentation files" FILES ${doc_files})
54 
55  IF(CONFIG_OPEN_DOCUMENTATION_FILE)
56  IF("${TARGET_OS}" STREQUAL "Windows")
57  SET(OPEN_INDEX_FILE START /B "${CMAKE_CURRENT_BINARY_DIR}/${CONFIG_DOCUMENTATION_FOLDER_NAME}/html/index.html") # async start
58  ELSE()
59  SET(OPEN_INDEX_FILE "") # untests for other os systems
60  ENDIF()
61  ELSE()
62  SET(OPEN_INDEX_FILE "")
63  ENDIF()
64 
65  # configure doxyfile with cmake variables
66  configure_file(${CMAKE_CURRENT_SOURCE_DIR}/doc/Doxyfile.in ${CMAKE_CURRENT_BINARY_DIR}/Doxyfile @ONLY)
67 
68  # add the target
69  add_custom_target(Documentation
70  ${DOXYGEN_EXECUTABLE} ${CMAKE_CURRENT_BINARY_DIR}/Doxyfile
71  COMMAND ${OPEN_INDEX_FILE} # will be empty and do nothing by default
72  WORKING_DIRECTORY ${CMAKE_CURRENT_BINARY_DIR}
73  COMMENT "Generating documentation" VERBATIM
74  SOURCES ${doc_files}
75  )
76  ENDIF()
77 ENDIF()

17 # Split arguments passed to a function into several lists separated by
18 # specified identifiers that do not have an associated list e.g.:
19 #
20 # SET(arguments
21 # hello world
22 # LIST3 foo bar
23 # LIST1 fuz baz
24 # )
25 # ARGUMENT_SPLITTER("${arguments}" "LIST1 LIST2 LIST3" ARG)
26 #
27 # results in 8 distinct variables:
28 # * ARG_DEFAULT_FOUND: 1
29 # * ARG_DEFAULT: hello;world
30 # * ARG_LIST1_FOUND: 1
31 # * ARG_LIST1: fuz;baz
32 # * ARG_LIST2_FOUND: 0
33 # * ARG_LIST2:
34 # * ARG_LIST3_FOUND: 1
35 # * ARG_LIST3: foo;bar

60  IF(NOT DEFINED THIRD_PARTY_DIR)

65 
66  INTERNAL_ARGUMENT_SPLITTER("${ARGN}" "LIBNAMES LIBDIRS INCLUDE_DIRS BINARY_INCLUDE_DIRS ABSOLUTE_INCLUDE_DIRS URL CHECKSUM SOURCE_DIR CMAKE_ARGUMENTS DEPENDENT_DEFINITIONS DEPENDENT_DEBUG_DEFINITIONS DEPENDENT_RELEASE_DEFINITIONS INSTALL CONFIG_FILE REQUIRED_PACKAGES USE_LIBRARY_OUTPUT_DIRECTORY USE_RUNTIME_OUTPUT_DIRECTORY" IAAP )
67 
68  IF(NOT "${IAAP_URL}" STREQUAL "")
69  SET(iaap_method DOWNLOAD_DIR "${THIRD_PARTY_DIR}/${iaap_name}"
70  URL "${IAAP_URL}"
71  URL_MD5 "${IAAP_CHECKSUM}")
72 
73  ELSEIF(NOT "${IAAP_SOURCE_DIR}" STREQUAL "")
74  INTERNAL_LIST_TO_STRING("${IAAP_SOURCE_DIR}" IAAP_CONVERTED_SOURCE_DIR)
75  SET(IAAP_SOURCE_DIR "${IAAP_CONVERTED_SOURCE_DIR}")
76  SET(iaap_method SOURCE_DIR "${IAAP_SOURCE_DIR}"

78 
79  ELSEIF(EXISTS "${THIRD_PARTY_DIR}/${iaap_name}")
80  SET(IAAP_SOURCE_DIR "${THIRD_PARTY_DIR}/${iaap_name}")
81  SET(iaap_method SOURCE_DIR "${IAAP_SOURCE_DIR}"

83  ENDIF()

85 
86  # Adds the packages that are required by the external cmake project to the cache variable "GLOBAL_CMAKE_PROJECTS_REQUIRED_PACKAGES"
87  IF(NOT "${IAAP_REQUIRED_PACKAGES}" STREQUAL "")
88  SET(CURRENT_MODULE_NAME "${iaap_name}")
89  SET(${iaap_name}_PACKAGES "" CACHE INTERNAL "")
90  FOREACH(package ${IAAP_REQUIRED_PACKAGES})
91  ACME_REQUIRED_PACKAGE("${package}")
92  SET(${iaap_name}_PACKAGES ${${iaap_name}_PACKAGES} "${package}" CACHE INTERNAL "")
93  ENDFOREACH()
94  SET(GLOBAL_CMAKE_PROJECTS_REQUIRED_PACKAGES ${GLOBAL_CMAKE_PROJECTS_REQUIRED_PACKAGES} "${iaap_name}_PACKAGES" CACHE INTERNAL "global list of all required packages that are used by a specific external cmake project")
95  ENDIF()

98  IF(NOT "${iaap_method}" STREQUAL "")
99  SET(BUILD_${iaap_name} 1 CACHE BOOL "Use ${iaap_name}")
100  SET(GLOBAL_BUILD_PROJECT ${GLOBAL_BUILD_PROJECT} BUILD_${iaap_name} CACHE INTERNAL "global list of all build variables of the added cmake projekts")

102  SET(ilts_cmake_arguments "")

104  SET(ilts_cmake_arguments ${ilts_cmake_arguments} -D${ilts_cmake_argument})

108  SET(ilts_install_command "")
109  ENDIF()

111 
112  IF("${TARGET_OS}" STREQUAL "Windows")

114  ELSE()
115  SET(BUILD_TYPE /${CMAKE_BUILD_TYPE})
116  ENDIF()

118  IF("${IAAP_USE_LIBRARY_OUTPUT_DIRECTORY}" STREQUAL "")

120  ELSE()
121  SET(USE_LIBRARY_OUTPUT_DIRECTORY "${IAAP_USE_LIBRARY_OUTPUT_DIRECTORY}")
122  MESSAGE(STATUS "Using custom library output dir: ${USE_LIBRARY_OUTPUT_DIRECTORY}")
123  ENDIF()
124 
125  IF("${IAAP_USE_RUNTIME_OUTPUT_DIRECTORY}" STREQUAL "")
126  SET(USE_RUNTIME_OUTPUT_DIRECTORY "${CMAKE_INSTALL_PREFIX}/${iaap_name}_${PROJECT_NAME}/bin/${TARGET_OS}_${TARGET_ARCH}${BUILD_TYPE}")
127  ELSE()
128  SET(USE_RUNTIME_OUTPUT_DIRECTORY "${IAAP_USE_RUNTIME_OUTPUT_DIRECTORY}")
129  ENDIF()
130  ExternalProject_Add(
131  ${iaap_name}
132  PREFIX ${GLOBAL_TOP_LEVEL_BINARY_DIR}/${iaap_name}_${PROJECT_NAME}
133  #PREFIX ${CMAKE_BINARY_DIR}/${iaap_name}
134  ${iaap_method}
135  UPDATE_COMMAND ""
136  CMAKE_ARGS -DCMAKE_TOOLCHAIN_FILE:PATH=${CMAKE_TOOLCHAIN_FILE}
137  -DCMAKE_CXX_FLAGS_RELEASE:STRING=${CMAKE_CXX_FLAGS_RELEASE}
138  -DCMAKE_C_FLAGS_RELEASE:STRING=${CMAKE_C_FLAGS_RELEASE}
139  -DCMAKE_CXX_FLAGS_DEBUG:STRING=${CMAKE_CXX_FLAGS_DEBUG}
140  -DCMAKE_C_FLAGS_DEBUG:STRING=${CMAKE_C_FLAGS_DEBUG}
141  -DCMAKE_CXX_FLAGS:STRING=${CMAKE_CXX_FLAGS}
142  -DCMAKE_C_FLAGS:STRING=${CMAKE_C_FLAGS}
143  -DCMAKE_BUILD_TYPE:STRING=${CMAKE_BUILD_TYPE}
144  -DCONFIG_BUILD_UNITTESTS:BOOLEAN=${CONFIG_BUILD_UNITTESTS}
145  -DCMAKE_ARCHIVE_OUTPUT_DIRECTORY:PATH=${USE_LIBRARY_OUTPUT_DIRECTORY}
146  -DCMAKE_LIBRARY_OUTPUT_DIRECTORY:PATH=${USE_LIBRARY_OUTPUT_DIRECTORY}
147  -DCMAKE_RUNTIME_OUTPUT_DIRECTORY:PATH=${USE_RUNTIME_OUTPUT_DIRECTORY}
148  -DCMAKE_HEADER_OUTPUT_DIRECTORY:PATH=${CMAKE_INSTALL_PREFIX}/${iaap_name}_${PROJECT_NAME}/include
149  -DCMAKE_RESOURCE_OUTPUT_DIRECTORY:PATH=${CMAKE_INSTALL_PREFIX}/${iaap_name}_${PROJECT_NAME}/res
150  -DCMAKE_DOC_OUTPUT_DIRECTORY:PATH=${CMAKE_INSTALL_PREFIX}/${iaap_name}_${PROJECT_NAME}/doc
151  #-DCMAKE_ARCHIVE_OUTPUT_DIRECTORY:PATH=${THIRD_PARTY_DIR}/deliverable/${iaap_name}/lib/${TARGET_OS}_${TARGET_ARCH}
152  #-DCMAKE_LIBRARY_OUTPUT_DIRECTORY:PATH=${THIRD_PARTY_DIR}/deliverable/${iaap_name}/lib/${TARGET_OS}_${TARGET_ARCH}
153  #-DCMAKE_INSTALL_PREFIX=${THIRD_PARTY_DIR}/deliverable/${iaap_name}
154  -DCMAKE_INSTALL_PREFIX:PATH=${CMAKE_INSTALL_PREFIX}
155  -DCONFIG_BUILD_GLOBAL_TEST_EXECUTABLE:BOOLEAN=${CONFIG_BUILD_GLOBAL_TEST_EXECUTABLE}
156  -DGLOBAL_TOP_LEVEL_SOURCE_DIR:PATH=${GLOBAL_TOP_LEVEL_SOURCE_DIR}
157  -DGLOBAL_TOP_LEVEL_BINARY_DIR:PATH=${GLOBAL_TOP_LEVEL_BINARY_DIR}
158  -DCONFIG_CREATE_DEPENDENCY_GRAPH:BOOL=${CONFIG_CREATE_DEPENDENCY_GRAPH}
159  -DCONFIG_CREATE_BUILD_REPORT:BOOL=${CONFIG_CREATE_BUILD_REPORT}
160  ${ilts_cmake_arguments}
161  "${ilts_install_command}"
162  )

164 # SET(${iaap_name}_DIR "${CMAKE_BINARY_DIR}/${iaap_name}")
165  SET(${iaap_name}_DIR "${IAAP_SOURCE_DIR}")
166 
167  SET(iacp_include_dirs "")
168  IF(NOT ${IAAP_URL} STREQUAL "")
169  FOREACH(iaap_include_dir ${IAAP_INCLUDE_DIRS})
170  SET(iacp_include_dirs "${iacp_include_dirs}" "${GLOBAL_TOP_LEVEL_BINARY_DIR}/${iaap_name}_${PROJECT_NAME}/${iaap_include_dir}")
171  ENDFOREACH()
172  ELSE()
173 
174  FOREACH(iaap_include_dir ${IAAP_INCLUDE_DIRS})
175  SET(iacp_include_dirs "${iacp_include_dirs}" "${CMAKE_INSTALL_PREFIX}/${iaap_name}_${PROJECT_NAME}/include/${iaap_include_dir}")
176  #SET(${iaap_name}_INCLUDE_DIR "${${iaap_name}_INCLUDE_DIR}" "${CMAKE_INSTALL_PREFIX}/include/${iaap_include_dir}")
177  ENDFOREACH()
178 
179  FOREACH(iaap_include_dir ${IAAP_BINARY_INCLUDE_DIRS})
180  SET(iacp_include_dirs "${iacp_include_dirs}" "${GLOBAL_TOP_LEVEL_BINARY_DIR}/${iaap_name}_${PROJECT_NAME}/src/${iaap_name}-build/${iaap_include_dir}")
181  endforeach()
182 
183  FOREACH(iaap_include_dir ${IAAP_ABSOLUTE_INCLUDE_DIRS})
184  SET(iacp_include_dirs "${iacp_include_dirs}" "${iaap_include_dir}")
185  ENDFOREACH()
186  ENDIF()
187 
188  INTERNAL_ADD_EXTERNAL_LIBRARY( ${iaap_name}
189  INCLUDE_DIRS "${iacp_include_dirs}"
190  LIBRARY_DIRS "${USE_LIBRARY_OUTPUT_DIRECTORY}" ${IAAP_LIBDIRS}
191  LIBNAMES "${IAAP_LIBNAMES}"
192  DEPENDENT_DEFINITIONS "${IAAP_DEPENDENT_DEFINITIONS}"
193  DEPENDENT_DEBUG_DEFINITIONS "${IAAP_DEPENDENT_DEBUG_DEFINITIONS}"
194  DEPENDENT_RELEASE_DEFINITIONS "${IAAP_DEPENDENT_RELEASE_DEFINITIONS}")
195 
196  ENDIF()
197  ENDIF()
198 ENDFUNCTION(INTERNAL_ADD_CMAKE_PROJECT)

201 FUNCTION(INTERNAL_ADD_EXTERNAL_LIBRARY ial_name)
202  INTERNAL_ARGUMENT_SPLITTER("${ARGN}" "INCLUDE_DIRS LIBRARY_DIRS LIBNAMES DEPENDENT_DEFINITIONS DEPENDENT_DEBUG_DEFINITIONS DEPENDENT_RELEASE_DEFINITIONS" IAL)
203 
204  SET(GLOBAL_CMAKE_PROJECTS ${GLOBAL_CMAKE_PROJECTS} "${ial_name}" CACHE INTERNAL "global list of all added cmake project")
205  SET(${ial_name}_INCLUDE_DIRS "" CACHE INTERNAL "")
206  FOREACH(ial_include_dir ${IAL_INCLUDE_DIRS})
207  SET(${ial_name}_INCLUDE_DIRS ${${ial_name}_INCLUDE_DIRS} "${ial_include_dir}" CACHE INTERNAL "")
208  ENDFOREACH()
209 
210  SET(${ial_name}_LIBRARIES_DIR "" CACHE INTERNAL "")
211  FOREACH(ial_library_dir ${IAL_LIBRARY_DIRS})
212  SET(${ial_name}_LIBRARIES_DIR ${${ial_name}_LIBRARIES_DIR} "${ial_library_dir}" CACHE INTERNAL "")
213  ENDFOREACH()
214 
215  SEPARATE_ARGUMENTS(IAL_LIBNAMES)
216  SET(${ial_name}_LIBRARIES ${IAL_LIBNAMES} CACHE INTERNAL "")
217  SET(${ial_name}_DEPENDENT_DEBUG_DEFINITIONS "${IAL_DEPENDENT_DEFINITIONS}" "${IAL_DEPENDENT_DEBUG_DEFINITIONS}" CACHE INTERNAL "")
218  SET(${ial_name}_DEPENDENT_RELEASE_DEFINITIONS "${IAL_DEPENDENT_DEFINITIONS}" "${IAL_DEPENDENT_RELEASE_DEFINITIONS}" CACHE INTERNAL "")
219  SET(${ial_name}_FOUND 1 CACHE INTERNAL "")
220  SET(${ial_name}_INTERNAL 1 CACHE INTERNAL "")
221 
222  MARK_AS_ADVANCED(
223  ${ial_name}_INCLUDE_DIRS
224  ${ial_name}_LIBRARIES_DIR
225  ${ial_name}_LIBRARIES
226  ${ial_name}_DEPENDENT_DEBUG_DEFINITIONS
227  ${ial_name}_DEPENDENT_RELEASE_DEFINITIONS
228  ${ial_name}_FOUND
229  ${ial_name}_INTERNAL
230  )
231 
232  SET(GLOBAL_EXTERNAL_LIBRARY_INCLUDE_DIR ${GLOBAL_EXTERNAL_LIBRARY_INCLUDE_DIR} ${${ial_name}_INCLUDE_DIRS} CACHE INTERNAL "global list of all variables which indicate the include directory of external libraries")
233  SET(GLOBAL_EXTERNAL_LIBRARY_LIBRARIES_DIR ${GLOBAL_EXTERNAL_LIBRARY_LIBRARIES_DIR} ${${ial_name}_LIBRARIES_DIR} CACHE INTERNAL "global list of all variables which indicate the library directory of external libraries")
234  SET(GLOBAL_EXTERNAL_LIBRARY_LIBRARIES ${GLOBAL_EXTERNAL_LIBRARY_LIBRARIES} ${${ial_name}_LIBRARIES} CACHE INTERNAL "global list of all variables which indicate the libraries of external libraries")
235 
236  LIST(REMOVE_ITEM "${ial_name}_DEPENDENT_DEBUG_DEFINITIONS" "")
237  LIST(LENGTH "${ial_name}_DEPENDENT_DEBUG_DEFINITIONS" list_length)
238  IF(${list_length})
239  SET(GLOBAL_EXTERNAL_LIBRARY_DEPENDENT_DEBUG_DEFINITIONS ${GLOBAL_EXTERNAL_LIBRARY_DEPENDENT_DEBUG_DEFINITIONS} ${ial_name}_DEPENDENT_DEBUG_DEFINITIONS CACHE INTERNAL "global list of all variables which indicate the dependent debug definitions of external libraries")
240  ENDIF()
241 
242  LIST(REMOVE_ITEM "${ial_name}_DEPENDENT_RELEASE_DEFINITIONS" "")
243  LIST(LENGTH "${ial_name}_DEPENDENT_RELEASE_DEFINITIONS" list_length)
244  IF(${list_length})
245  SET(GLOBAL_EXTERNAL_LIBRARY_DEPENDENT_RELEASE_DEFINITIONS ${GLOBAL_EXTERNAL_LIBRARY_DEPENDENT_RELEASE_DEFINITIONS} ${ial_name}_DEPENDENT_RELEASE_DEFINITIONS CACHE INTERNAL "global list of all variables which indicate the dependent release definitions of external libraries")
246  ENDIF()
247 
248 
249  LIST(REMOVE_ITEM "${ial_name}_FOUND" "")
250  LIST(LENGTH "${ial_name}_FOUND" list_length)
251  IF(${list_length})
252  SET(GLOBAL_EXTERNAL_LIBRARY_FOUND ${GLOBAL_EXTERNAL_LIBRARY_FOUND} ${ial_name}_FOUND CACHE INTERNAL "global list of all variables which indicate if an external library was found")
253  ENDIF()
254 
255  LIST(REMOVE_ITEM "${ial_name}_INTERNAL" "")
256  LIST(LENGTH "${ial_name}_INTERNAL" list_length)
257  IF(${list_length})
258  SET(GLOBAL_EXTERNAL_LIBRARY_INTERNAL ${GLOBAL_EXTERNAL_LIBRARY_INTERNAL} ${ial_name}_INTERNAL CACHE INTERNAL "")
259  ENDIF()
260 
261  #SET(GLOBAL_INCLUDE_DIRECTORIES ${GLOBAL_INCLUDE_DIRECTORIES} ${${ial_name}_INCLUDE_DIR} CACHE INTERNAL "collect include directories")
262  #SET(GLOBAL_LIB_DIRECTORIES ${GLOBAL_LIB_DIRECTORIES} ${${ial_name}_LIBRARIES_DIR} CACHE INTERNAL "collect lib directories")
263  #SET(GLOBAL_LIBRARIES ${GLOBAL_LIBRARIES} ${${ial_name}_LIBRARIES} CACHE INTERNAL "collect all linkable libraries")
264 
265  #message("include dirs ${${ial_name}_INCLUDE_DIR}")
266  #message("library dirs ${${ial_name}_LIBRARIES_DIR}")
267  #message("libraries ${${ial_name}_LIBRARIES}")
268  #message("depandent defs ${${ial_name}_DEPENDENT_DEFINITIONS}")
269 
270 ENDFUNCTION(INTERNAL_ADD_EXTERNAL_LIBRARY)

338 FUNCTION(INTERNAL_ADD_MODULE ad_module_name ad_type)
339  INTERNAL_ADD_MODULE_INTERNAL(${ad_module_name} ${ad_type})
340  STRING(TOUPPER ${ad_module_name} CURRENT_UPPER_MODULE_NAME)
341  SET(WITH_${CURRENT_UPPER_MODULE_NAME} 1 CACHE INTERNAL "Use module ${CURRENT_UPPER_MODULE_NAME}")
342  SET(GLOBAL_WITH_MODULE ${GLOBAL_WITH_MODULE} WITH_${CURRENT_UPPER_MODULE_NAME} CACHE INTERNAL "global list of all used modules")
343 
344  IF("${ad_type}" STREQUAL "static")
345  SET(GLOBAL_UTILS_MODULES_STATIC ${GLOBAL_UTILS_MODULES_STATIC} "${ad_module_name}" CACHE INTERNAL "stores the module names of all static modules")
346  ENDIF()
347 
348  IF("${ad_type}" STREQUAL "dynamic")
349  SET(GLOBAL_UTILS_MODULES_DYNAMIC ${GLOBAL_UTILS_MODULES_DYNAMIC} "${ad_module_name}" CACHE INTERNAL "stores the module names of all dynamic modules")
350  ENDIF()
351 
352  IF("${ad_type}" STREQUAL "exe")
353  SET(GLOBAL_UTILS_MODULES_EXE ${GLOBAL_UTILS_MODULES_EXE} "${ad_module_name}" CACHE INTERNAL "stores the module names of all exe modules")
354  ENDIF()
355 ENDFUNCTION(INTERNAL_ADD_MODULE)

17 FUNCTION(INTERNAL_ADD_COMPILER_FLAGS_TO_TARGET acf_target)
18  INTERNAL_LIST_TO_STRING("${ARGN}" ARGN) # CMake uses semicolon-separated lists almost everywhere except in variables or properties
19  # whose names contain "FLAGS" (which predate the semicolon list convention).
20  # There space-separated lists are used.
21  GET_TARGET_PROPERTY(current_compile_flags ${acf_target} COMPILE_FLAGS)
22  IF(current_compile_flags STREQUAL "current_compile_flags-NOTFOUND")
23  SET(current_compile_flags "")
24  ELSE()
25  SET(current_compile_flags "${current_compile_flags} ") # Seperate new compiler flags from the existing ones
26  ENDIF()
27  SET(current_compile_flags "${current_compile_flags}${ARGN}" )
28 
29  SET_TARGET_PROPERTIES(${acf_target} PROPERTIES COMPILE_FLAGS "${current_compile_flags}" )
30 ENDFUNCTION(INTERNAL_ADD_COMPILER_FLAGS_TO_TARGET)

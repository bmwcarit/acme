57 FUNCTION(INTERNAL_EXPORT_TARGET)
58  IF(NOT ${ARGV0} STREQUAL "")
59  SET(targetDirectoryPath "${GLOBAL_TOP_LEVEL_BINARY_DIR}/export/${PROJECT_NAME}/${ARGV0}")
60  ELSE()
61  SET(targetDirectoryPath "${GLOBAL_TOP_LEVEL_BINARY_DIR}/export/${PROJECT_NAME}")
62  ENDIF()
63  SET(${CURRENT_MODULE_NAME}_TARGETDIRS ${${CURRENT_MODULE_NAME}_TARGETDIRS} ${targetDirectoryPath} CACHE INTERNAL "")
64  SET(TARGETS_TO_COPY ${TARGETS_TO_COPY} ${CURRENT_MODULE_NAME} CACHE INTERNAL "")
65 ENDFUNCTION(INTERNAL_EXPORT_TARGET)

68 FUNCTION(INTERNAL_EXPORT_DIRECTORY sourceDir) # optional parameter targetdir

72  SET(targetDirectoryPath "${GLOBAL_TOP_LEVEL_BINARY_DIR}/export/${PROJECT_NAME}/${sourceDir}/")

93 # collect everything for tar building
94 FUNCTION(INTERNAL_BUILD_RELEASE_TAR)
95  MESSAGE(STATUS "Adding Tarfile export target")
96  # the top level target for building the tar
97  add_custom_target(ExportTar COMMENT "Exporting marked Files into a tar" )
98 
99  # make files to copy out of targets to copy
100  FOREACH(TARGET_TO_COPY ${TARGETS_TO_COPY})
101  get_target_property(fullFilePath ${TARGET_TO_COPY} LOCATION)
102  get_filename_component(FILENAME "${fullFilePath}" NAME)
103  FOREACH(TARGET_DIR ${${TARGET_TO_COPY}_TARGETDIRS})
104  add_custom_command(TARGET ExportTar COMMAND ${CMAKE_COMMAND} -E copy_if_different "${fullFilePath}" "${TARGET_DIR}/${FILENAME}")
105  ENDFOREACH()
106  ENDFOREACH()
107 
108  # go through all directories to be copied
109  FOREACH(DIRECTORY_TO_COPY ${DIRECTORIES_TO_COPY})
110  FOREACH(TARGET_DIR ${${DIRECTORY_TO_COPY}_TARGETDIRS})
111  add_custom_command(TARGET ExportTar COMMAND ${CMAKE_COMMAND} -E copy_directory "${DIRECTORY_TO_COPY}" "${TARGET_DIR}")
112  ENDFOREACH()
113  ENDFOREACH()
114 
115  # go through all files to include in tar
116  FOREACH(FILE_TO_COPY ${FILES_TO_COPY})
117  FOREACH(TARGET_PATH ${${FILE_TO_COPY}_TARGETPATHS})
118  add_custom_command(TARGET ExportTar COMMAND ${CMAKE_COMMAND} -E copy_if_different "${FILE_TO_COPY}" "${TARGET_PATH}")
119  ENDFOREACH()
120  ENDFOREACH()
121 
122  # tar everything beneath build/export dir
123  add_custom_command(TARGET ExportTar COMMAND ${CMAKE_COMMAND} -E tar cvf ${PROJECT_NAME}.tar "${PROJECT_NAME}" WORKING_DIRECTORY ${CMAKE_CURRENT_BINARY_DIR}/export/)
124 
125  # copy tar to deliverable folder
126  add_custom_command(TARGET ExportTar COMMAND ${CMAKE_COMMAND} -E copy_if_different "${CMAKE_CURRENT_BINARY_DIR}/export/${PROJECT_NAME}.tar" "${CMAKE_INSTALL_PREFIX}/${PROJECT_NAME}.tar")
127 ENDFUNCTION(INTERNAL_BUILD_RELEASE_TAR)

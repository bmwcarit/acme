49 FUNCTION(INTERNAL_CREATE_TEST_COVERAGE)
50  IF(CONFIG_CREATE_TEST_COVERAGE)
51  IF(NOT "${LCOV_PATH}" STREQUAL "LCOV_PATH-NOTFOUND")
52  MESSAGE("")
53  MESSAGE(---------------------------------------------------------------------------)
54  MESSAGE("Creating test coverage...")
55 
56  IF("${TARGET_OS}" STREQUAL "Linux")
57  INTERNAL_TEST_COVERAGE_LINUX()
58  ELSEIF("${TARGET_OS}" STREQUAL "Windows")
59  MESSAGE(STATUS "The feature is currently just available on Linux")
60  ELSE()
61 
62  ENDIF()
63  ENDIF()
64  ENDIF()
65 ENDFUNCTION(INTERNAL_CREATE_TEST_COVERAGE)
66 
67 
68 FUNCTION(INTERNAL_TEST_COVERAGE_LINUX)
69  SET(settings_correct 1)
70 
71  IF("${GLOBAL_UTILS_MODULES_TESTS}" STREQUAL "")
72  MESSAGE(STATUS "WARNING: No test modules have been found.")
73  SET(settings_correct 0)
74 
75  ELSE()
76  IF("${LCOV_PATH}" STREQUAL "LCOV_PATH-NOTFOUND")
77  MESSAGE(STATUS "WARNING: The required tool lcov is not installed. Install \"lcov\" to create the test coverage target")
78  SET(settings_correct 0)
79  ENDIF()
80 
81  IF(CONFIG_BUILD_GLOBAL_TEST_EXECUTABLE)
82  MESSAGE(STATUS "WARNING: The cache variable \"CONFIG_BUILD_GLOBAL_TEST_EXECUTABLE\" ist TRUE. To create the target \"Test_Coverage\" DISABLE this cache variable.")
83  SET(settings_correct 0)
84  ENDIF()
85 
86  IF(NOT CONFIG_BUILD_UNITTESTS)
87  MESSAGE(STATUS "WARNING: The cache variable \"CONFIG_BUILD_UNITTESTS\" ist FALSE. To create the target \"Test_Coverage\" ENABLE this cache variable.")
88  SET(settings_correct 0)
89  ENDIF()
90 
91  IF(NOT "${CMAKE_BUILD_TYPE}" STREQUAL "Debug")
92  MESSAGE(STATUS "WARNING: The cache variable \"CMAKE_BUILD_TYPE\" ist not \"Debug\". To create the target \"Test_Coverage\" set this cache variable to \"Debug\".")
93  SET(settings_correct 0)
94  ENDIF()
95  ENDIF()
96 
97 
98  IF("${settings_correct}" STREQUAL "0")
99  MESSAGE(STATUS "WARNING: The target \"Test_Coverage\" was not created.")
100  ELSE()
101  MESSAGE(STATUS "All necessary settings are correct. A Test-Coverage-Target has been created.")
102  MESSAGE(STATUS "Use \"make Test_Coverage\" after building and installing the project in the build-directory to generate the test coverage.")
103 
104  SET(path_to_testexe "${CMAKE_RUNTIME_OUTPUT_DIRECTORY}")
105  SET(path_to_test_coverage "${CMAKE_DOC_OUTPUT_DIRECTORY}/test_coverage")
106  SET(path_to_build_dir "${GLOBAL_TOP_LEVEL_BINARY_DIR}")
107 
108  SET(GCC_COVERAGE_COMPILE_FLAGS "-fprofile-arcs -ftest-coverage")
109  SET(GCC_COVERAGE_LINK_FLAGS "-fprofile-arcs")
110 
111  FOREACH(module_name ${GLOBAL_MODULE_NAMES})
112  STRING(TOUPPER ${module_name} upper_module_name)
113  IF(${WITH_${upper_module_name}} AND ${${module_name}_BUILD_ENABLED})
114  INTERNAL_ADD_COMPILER_FLAGS_TO_TARGET("${module_name}" "${GCC_COVERAGE_COMPILE_FLAGS}")
115  INTERNAL_ADD_DEBUG_LINKER_FLAGS_TO_TARGET("${module_name}" "${GCC_COVERAGE_LINK_FLAGS}")
116  ENDIF()
117  ENDFOREACH()
118 
119  FOREACH(test_name ${GLOBAL_UTILS_MODULES_TESTS})
120  INTERNAL_ADD_COMPILER_FLAGS_TO_TARGET("${test_name}" "${GCC_COVERAGE_COMPILE_FLAGS}")
121  INTERNAL_ADD_DEBUG_LINKER_FLAGS_TO_TARGET("${test_name}" "${GCC_COVERAGE_LINK_FLAGS}")
122  ENDFOREACH()
123 
124  IF(NOT IS_DIRECTORY ${path_to_test_coverage})
125  FILE(MAKE_DIRECTORY ${path_to_test_coverage})
126  ENDIF()
127 
128  IF(EXISTS ${ACME_PATH}/plugins/BuildReport/2_test_coverage_template.dox)
129  CONFIGURE_FILE(${ACME_PATH}/plugins/BuildReport/2_test_coverage_template.dox "${CMAKE_DOC_OUTPUT_DIRECTORY}/")
130  ENDIF()
131 
132  ADD_CUSTOM_TARGET(Test_Coverage
133  COMMENT "Creating target for test coverage..."
134  VERBATIM)
135 
136  ADD_DEPENDENCIES(Test_Coverage ${GLOBAL_UTILS_MODULES_TESTS})
137 
138  FOREACH(entry ${GLOBAL_UTILS_MODULES_TESTS})
139  ADD_CUSTOM_COMMAND(TARGET Test_Coverage
140  POST_BUILD
141  COMMAND lcov
142  ARGS -d ${path_to_build_dir} --zerocounters
143  COMMAND lcov
144  ARGS -c -i -d ${path_to_build_dir}/modules -o base_${entry}.info
145  WORKING_DIRECTORY "${path_to_test_coverage}"
146  COMMENT "Initializing code coverage counters..."
147  )
148 
149  ADD_CUSTOM_COMMAND(TARGET Test_Coverage
150  COMMAND ./${entry}
151  WORKING_DIRECTORY "${path_to_testexe}"
152  COMMENT "Executing test executables..."
153  )
154 
155  ADD_CUSTOM_COMMAND(TARGET Test_Coverage
156  COMMAND lcov
157  ARGS -c -d ${path_to_build_dir}/modules -o test_${entry}.info
158  COMMAND lcov
159  ARGS -a base_${entry}.info -a test_${entry}.info -o total_${entry}.info
160  WORKING_DIRECTORY "${path_to_test_coverage}"
161  COMMENT "Generating test coverage file for module ${entry}..."
162  )
163  ENDFOREACH()
164 
165  SET(collect_coverage_files "")
166  FOREACH(entry ${GLOBAL_UTILS_MODULES_TESTS})
167  SET(collect_coverage_files ${collect_coverage_files} -a total_${entry}.info)
168  ENDFOREACH()
169  SET(collect_coverage_files ${collect_coverage_files} -o total_${PROJECT_NAME}.info)
170 
171  ADD_CUSTOM_COMMAND(TARGET Test_Coverage
172  POST_BUILD
173  COMMAND lcov
174  ARGS ${collect_coverage_files}
175  COMMAND genhtml
176  ARGS -s -l total_${PROJECT_NAME}.info
177  WORKING_DIRECTORY "${path_to_test_coverage}"
178  COMMENT "Merging test coverage files of the testmodules and creating the html output of the test coverage..."
179  )
180 
181  ENDIF()
182 ENDFUNCTION(INTERNAL_TEST_COVERAGE_LINUX)

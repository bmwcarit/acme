66 FUNCTION(INTERNAL_CPPCHECK)
67  IF(CONFIG_CREATE_CPPCHECK)
68  IF(CPPCHECK_FOUND)
69  MESSAGE(STATUS "Finalize cppcheck for targets: ${TARGETS_TO_CHECK}")
70  add_custom_target(cppcheck COMMENT "static source code analysis with cppcheck")
71  add_custom_target(cppcheck_xml COMMENT "static source code analysis with cppcheck (xml export)")
72 
73  FOREACH(TARGET_TO_CHECK ${TARGETS_TO_CHECK})
74  get_target_property(cppcheckSourceFiles ${TARGET_TO_CHECK} SOURCES)
75 
76  # get include dirs of the target
77  get_target_property(cppcheckIncludeDirs ${TARGET_TO_CHECK} INCLUDE_DIRECTORIES)
78  set( cppcheckIncludes )
79  foreach(includeDir ${cppcheckIncludeDirs} )
80  LIST (APPEND cppcheckIncludes "-I${includeDir}")
81  endforeach()
82 
83  # get compile definitions
84 
85  IF(NOT "${CMAKE_BUILD_TYPE}" STREQUAL "")
86  STRING(TOUPPER ${CMAKE_BUILD_TYPE} buildTypeUpper)
87  ENDIF()
88  IF ("${buildTypeUpper}" STREQUAL "")
89  SET(buildTypeUpper "DEBUG")
90  ENDIF()
91  get_target_property(cppcheckDefinesRaw ${TARGET_TO_CHECK} COMPILE_DEFINITIONS_${buildTypeUpper})
92  set( cppcheckDefines )
93  foreach(cppcheckDefine ${cppcheckDefinesRaw} )
94  LIST (APPEND cppcheckDefines "-D${cppcheckDefine}")
95  endforeach()
96 
97  add_custom_command(TARGET cppcheck
98  COMMAND ${CPPCHECK_EXECUTABLE}
99  ARGS -q --inline-suppr --enable=all ${cppcheckIncludes} ${cppcheckDefines} ${cppcheckSourceFiles}
100  )
101  add_custom_command(TARGET cppcheck_xml
102  COMMAND ${CPPCHECK_EXECUTABLE}
103  ARGS --xml-version=2 -q --inline-suppr --enable=all ${cppcheckIncludes} ${cppcheckDefines} ${cppcheckSourceFiles}
104  )
105 
106  ENDFOREACH()
107  ENDIF()
108  ENDIF()
109 ENDFUNCTION(INTERNAL_CPPCHECK)
